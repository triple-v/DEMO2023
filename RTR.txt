en  //  Вход в привилегированный режим

(ip route 0.0.0.0 0.0.0.0 4.4.4.1)    // Под вопросом



int g1  // Входим в настройки первого интерфейса сетевой карты

ip addr 4.4.4.100 255.255.255.0   // Настройка IP адресации первого интерфейса

no shutdown   // Включаем запуск интерфейса при последующей загрузке ВМ

exit    // Выходим из настроек первого интерфейса



int g2    // Входим в настройки второго интерфейса сетевой карты

ip addr 192.168.100.254 255.255.255.0   // Настройка IP адресации второго интерфейса

no shutdown   // Включаем запуск интерфейса при последующей загрузке ВМ

exit    // Выходим из настроек второго интерфейса



router eigrp 6500   // Входим в настройку сетевого адаптера

(network 192.168.100.0 0.0.0.255)   // Под вопросом

(network 172.16.1.0 0.0.0.255)    // Под вопросом

no log-neighbor-warnings    // Отключаем вывод предупреждений в консоли, мешающих вводу команд

exit    // Выходим из настроек сетевого адаптера



int Tunnel1   // Вход в настройка туннеля

ip addr 172.16.1.1 255.255.255.0    // Придумываем IP адрес туннеля для первой ВМ (первые три числа должны совпадать с адресом второй ВМ)

tunnel source 4.4.4.100   //  Задаём IP первого интерфейса текущей ВМ в качестве начала туннеля

tunnel destination 5.5.5.100    // Задаём IP первого интерфейса параллельной ВМ в качестве конца туннеля

tunnel mode gre ip    // Задаём режим туннелирования

exit    // Выходим из настроек туннелирования
